def index
  hard_sort = %w(io_interface tms core upload media_content media_object copyright zhdk_bereich media_set)
  @contexts = MetaContext.all.sort {|a,b| (hard_sort.index(a.name) || a.id) <=> (hard_sort.index(b.name) || b.id) }
end
