def index
  people = Person.search(params[:term])
  users = people.map(&:user).compact
  
  if params[:group_id]
    group = Group.find(params[:group_id])
    users -= group.users
  end
  
  respond_to do |format|
    format.html
    format.js { render :json => users.map {|x| {:id => x.id, :value => x.to_s} } }
  end
end
